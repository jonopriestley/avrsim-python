; Compares whether n1 + n2 == condition or if its lower of higher

.section .data

n1: .byte 12
n2: .byte 33
condition: .byte 45
w: .byte 0


.section .text
    .global asm_function

asm_function:
    lds r18, n1
    lds r19, n2
    add r18, r19
    lds r19, condition
    cp r18, r19
    breq equal
    brge greater
    ldi r26, -1
    ldi r27, -1
    ret

equal:
    clr r26
    clr r27
    ret

greater:
    clr r27
    ldi r26, 1
    ret


.end
